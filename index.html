chatWorker.onmessage = function(event) { const jsonArray = event.data; statusDiv.innerText = `Success! Parsed ${jsonArray.length} messages.`; const chatDisplay = document.getElementById('chat-display'); chatDisplay.innerHTML = ""; // Clear out any old messages // Loop through every message the worker sent back jsonArray.forEach(msg => { // Create a new div for the message const msgBubble = document.createElement('div'); // Add some quick CSS styling to make it look like a chat bubble msgBubble.style.backgroundColor = "#fff"; msgBubble.style.padding = "8px"; msgBubble.style.margin = "5px 0"; msgBubble.style.borderRadius = "5px"; msgBubble.style.boxShadow = "0 1px 1px rgba(0,0,0,0.1)"; // Format the text inside the bubble msgBubble.innerHTML = ` ${msg.author}
${msg.text}
${msg.date} - ${msg.time}
`; // Add the bubble to the screen chatDisplay.appendChild(msgBubble); }); };
